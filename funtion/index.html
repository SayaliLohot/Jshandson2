while loop-
    
        function factorialize(num) {
            var result = num;
            if (num === 0 || num === 1) 
              console.log(1); 
            while (num > 1) { 
              num--;
              result *= num;
            }
            console.log(result);
          }
          factorialize(5);
Result-120 
for loop-
            function factorialize(num) {
              if (num === 0 || num === 1)
                console.log(1);
              for (var i = num - 1; i >= 1; i--) {
                num *= i;
              }
             console.log(num);
            }
            factorialize(4);
Result-24
